use bdVenda;

create table tbPereciveis(
id int primary key auto_increment
,Produto varchar(40)
,valor decimal(5,2)
,qtd int
);

 select * from tbPereciveis;

 
create table tbItemVendido(
id int primary key auto_increment
,ProdutoVendido varchar (50)
,ValorVendido decimal (5,2)
,qtdVendida int
);
